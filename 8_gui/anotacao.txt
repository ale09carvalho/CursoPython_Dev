O Flet permite que desenvolvedores criem facilmente aplicativos web, mobile e desktop em tempo real em Python.
A Flet UI foi desenvolvida com Flutter, para que seu aplicativo tenha uma aparência profissional e possa ser entregue em qualquer plataforma. 
A Flet simplifica o modelo do Flutter combinando "widgets" menores em "controles" prontos para uso com um modelo de programação imperativa.

def main(page: ft.Page):
    
    # Configurações da página
    page.add(
        # Área segura - cabeça 
        ft.SafeArea(
            ft.Container(
                ft.Text("Ola, Mundo flet Python!!! 🎶🐍🐍\n")
            ), 
            expand=True,
        )
    )

ft.app(target=main)

import flet as ft


def main(page: ft.Page):
    
    # Configurações da página
    page.add(
        ft.SafeArea(  # Cabeçalho
            ft.Column(
                [
                    ft.Container(
                        ft.Text(
                            "Ola, Mundo flet Python!!!",
                            size=30,
                            weight="bold",
                        ),
                        expand=True,
                    ),
                    # Corpo
                    ft.Text("😎👻👽 Rodapé"),
                ]
            )
        )
    )

ft.app(target=main)
